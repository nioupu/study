import csv 模块

    if open_ports:
        csv_filename = f"{target_host}_开放端口扫描结果.csv"
        with open(csv_filename, "w", newline="", encoding="utf-8") as csv_file:
            writer = csv.writer(csv_file)
            writer.writerow(["目标主机", "开放端口"])
            for port in open_ports:
                writer.writerow([target_host, port])
                print(f"\n【CSV导出完成】结果已保存到：{csv_filename}")
